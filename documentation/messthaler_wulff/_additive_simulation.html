395    @staticmethod
396    def points_within_dist(n, dist):
397        if n == 0:
398            yield ()
399            return
400        if n == 1:
401            for i in list(range(-dist, dist + 1)):
402                yield (i,)
403            return
404
405        for i in range(-dist, dist + 1):
406            for p in SimpleNeighborhood.points_within_dist(n - 1, dist):
407                yield [*p, i]
408
409    @staticmethod
410    def points_with_dist(n, dist):
411        if dist == 0:
412            yield [0] * n
413            return
414        if dist == 1:
415            for p in SimpleNeighborhood.points_within_dist(n, dist):
416                if all(x == 0 for x in p): continue
417                yield p
418            return
419
420        for i in range(n):
421            for sign in [-1, 1]:
422                for p in SimpleNeighborhood.points_within_dist(n - 1, dist):
423                    yield *p[:i], sign * dist, *p[i:]
424
425    def find_neighbors(self):
426        for dist in range(1, 10):
427            changed = 0
428
429            for p in SimpleNeighborhood.points_with_dist(self.n, dist):
430                point = np.dot(self.transform, np.array(p))
431                if np.linalg.norm(point) <= 1:
432                    p = tuple(p)
433                    if p not in self.base_neighborhood:
434                        self.base_neighborhood.add(p)
435                        changed += 1
436
437            if changed == 0:
438                return
439
440        raise RuntimeError("Could not find all neighbors")

442    def __call__(self, pos):
443        return [move(pos, x) for x in self.base_neighborhood]
